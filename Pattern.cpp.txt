#include <iostream>
using namespace std;
int main()
{
    int n;
    cin>>n;
    int k=1;
    for(int i=1;i<=2*n;i=i+2,k++)
    {
        for(int j=1;j<=(2*n-k);j++)
        {
            cout<<" ";
        }
        for(int j=i;j>=1;j--)
        {
            cout<<i<<" ";
        }
        cout<<endl;
    }
    return 0;
}
